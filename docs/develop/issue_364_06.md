def update_description_with_features(
    metadata_path: Path,
    invoice_path: Path,
    description_key: str = "datasetDescription",
) -> None:
    """Update invoice description field with feature information from metadata.

    Args:
        metadata_path: Path to metadata JSON file.
        invoice_path: Path to invoice JSON file to update.
        description_key: Key in invoice to update (default: "datasetDescription").

    Raises:
        FileNotFoundError: If metadata or invoice files don't exist.
        ValueError: If description_key not found in invoice.

    Example:
        update_description_with_features(
            metadata_path=Path("metadata.json"),
            invoice_path=Path("invoice.json"),
            description_key="datasetDescription"
        )
    """
    # Implementation from lines 1023-1092
    from rdetoolkit.invoicefile._invoice import InvoiceFile

    # Load metadata
    if not metadata_path.exists():
        emsg = f"Metadata file not found: {metadata_path}"
        raise FileNotFoundError(emsg)

    metadata_obj = readf_json(metadata_path)

    # Load invoice
    invoice_file = InvoiceFile(invoice_path)

    # Collect feature values
    collected_values = __collect_values_from_variable("features", metadata_obj)

    # Format description entries
    description_entries = []
    for value in collected_values:
        entry = __format_description_entry(
            name_ja=value.get("name_ja", ""),
            header_value=value.get("value"),
            unit=value.get("unit"),
        )
        description_entries.append(entry)

    # Update invoice
    current_description = invoice_file.invoice_obj.get("basic", {}).get(description_key, "")
    updated_description = current_description + "\n" + "\n".join(description_entries)
    invoice_file.invoice_obj["basic"][description_key] = updated_description.strip()

    # Save
    invoice_file.overwrite()
    logger.info(f"Description updated in {invoice_path}")

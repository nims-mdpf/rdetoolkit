"""Comprehensive backward compatibility tests for invoicefile package."""
from __future__ import annotations

    """Verify all public exports are available from main module."""
    # Classes
    from rdetoolkit.invoicefile import (
        ExcelInvoiceFile,
        ExcelInvoiceTemplateGenerator,
        InvoiceFile,
        MagicVariableResolver,
        RuleBasedReplacer,
        SmartTableFile,
        TemplateGenerator,
    )

    # Functions
    from rdetoolkit.invoicefile import (
        apply_default_filename_mapping_rule,
        apply_magic_variable,
        backup_invoice_json_files,
        check_exist_rawfiles,
        check_exist_rawfiles_for_folder,
        overwrite_invoicefile_for_dpfterm,
        read_excelinvoice,
        update_description_with_features,
    )

    # Constants
    from rdetoolkit.invoicefile import (
        EX_GENERALTERM,
        EX_SPECIFICTERM,
        MAGIC_VARIABLE_PATTERN,
        STATIC_DIR,
        logger,
    )

    # All imports successful
    assert InvoiceFile is not None
    assert ExcelInvoiceFile is not None
    assert SmartTableFile is not None

```python
# src/rdetoolkit/invoicefile/_magic_variable.py
"""Magic variable expansion for invoice files.

This module provides classes and functions for resolving magic variables
(e.g., ${variable_name}) in invoice files using metadata from datasets.
"""
from __future__ import annotations

import re
from pathlib import Path
from typing import TYPE_CHECKING, Any

from rdetoolkit.fileops import readf_json
from rdetoolkit.invoicefile._helpers import MAGIC_VARIABLE_PATTERN, logger

if TYPE_CHECKING:
    from rdetoolkit.models.rde2types import RdeDatasetPaths


def _load_metadata(dataset_paths: RdeDatasetPaths | None) -> dict[str, Any] | None:
    """Load metadata from dataset paths.

    Args:
        dataset_paths: Dataset paths containing metadata.json location.

    Returns:
        Metadata dictionary if found, None otherwise.
    """
    # Implementation from lines 1448-1466
    if dataset_paths is None:
        return None

    # Try to find metadata.json
    metadata_candidates = [
        dataset_paths.struct / "metadata.json",
        dataset_paths.main / "metadata.json",
    ]

    for metadata_path in metadata_candidates:
        if metadata_path.exists():
            logger.debug(f"Loading metadata from {metadata_path}")
            return readf_json(metadata_path)

    logger.debug("No metadata.json found in dataset paths")
    return None


class MagicVariableResolver:
    """Resolver for magic variables in strings.

    This class resolves magic variable expressions like ${variable_name}
    by looking up values in metadata dictionaries.

    Attributes:
        metadata (dict[str, Any] | None): Metadata dictionary for variable resolution.

    Example:
        metadata = {"constant": {"sample_id": {"value": "12345"}}}
        resolver = MagicVariableResolver(metadata)
        result = resolver.resolve("Sample: ${sample_id}")
        # result: "Sample: 12345"
    """

    def __init__(self, metadata: dict[str, Any] | None = None):
        """Initialize resolver with metadata.

        Args:
            metadata: Metadata dictionary containing variable values.
        """
        self.metadata = metadata or {}

    def resolve(self, text: str) -> str:
        """Resolve magic variables in text.

        Args:
            text: Text containing magic variable expressions.

        Returns:
            Text with magic variables replaced by their values.
        """
        # Implementation from lines 1252-1415
        if not text or not isinstance(text, str):
            return text

        # Find all magic variable patterns
        matches = MAGIC_VARIABLE_PATTERN.findall(text)
        if not matches:
            return text

        result = text
        for var_name in matches:
            value = self._lookup_variable(var_name)
            if value is not None:
                pattern = re.escape(f"${{{var_name}}}")
                result = re.sub(pattern, str(value), result)
            else:
                logger.warning(f"Magic variable not found: ${{{var_name}}}")

        return result

    def _lookup_variable(self, var_name: str) -> Any | None:
        """Look up variable value in metadata.

        Args:
            var_name: Variable name to look up.

        Returns:
            Variable value if found, None otherwise.
        """
        # Search in metadata structure
        if not self.metadata:
            return None

        # Try constant variables
        if "constant" in self.metadata:
            constant_vars = self.metadata["constant"]
            if var_name in constant_vars:
                var_obj = constant_vars[var_name]
                if isinstance(var_obj, dict) and "value" in var_obj:
                    return var_obj["value"]
                return var_obj

        # Try variable variables
        if "variable" in self.metadata:
            variable_vars = self.metadata["variable"]
            if var_name in variable_vars:
                var_obj = variable_vars[var_name]
                if isinstance(var_obj, dict) and "value" in var_obj:
                    return var_obj["value"]
                return var_obj

        return None

    def resolve_dict(
        self, target_dict: dict[str, Any], keys: list[str] | None = None
    ) -> dict[str, Any]:
        """Resolve magic variables in dictionary values.

        Args:
            target_dict: Dictionary to process.
            keys: List of keys to process (if None, process all string values).

        Returns:
            Dictionary with magic variables resolved.
        """
        # Implementation details
        if keys is None:
            # Process all string values
            for key, value in target_dict.items():
                if isinstance(value, str):
                    target_dict[key] = self.resolve(value)
        else:
            # Process only specified keys
            for key in keys:
                if key in target_dict and isinstance(target_dict[key], str):
                    target_dict[key] = self.resolve(target_dict[key])

        return target_dict


def apply_magic_variable(
    invoice_file_path: Path,
    dataset_paths: RdeDatasetPaths | None = None,
    *,
    metadata: dict[str, Any] | None = None,
) -> None:
    """Apply magic variable resolution to invoice file.

    This function reads an invoice file, resolves any magic variables
    using metadata, and overwrites the file with resolved values.

    Args:
        invoice_file_path: Path to invoice JSON file to process.
        dataset_paths: Dataset paths containing metadata (optional).
        metadata: Explicit metadata dictionary (optional, overrides dataset_paths).

    Example:
        apply_magic_variable(
            invoice_file_path=Path("invoice.json"),
            dataset_paths=dataset_paths
        )
    """
    # Implementation from lines 1467-1527
    from rdetoolkit.invoicefile._invoice import InvoiceFile

    # Load metadata
    if metadata is None:
        metadata = _load_metadata(dataset_paths)

    if metadata is None:
        logger.debug("No metadata available for magic variable resolution")
        return

    # Load invoice
    invoice_file = InvoiceFile(invoice_file_path)

    # Create resolver
    resolver = MagicVariableResolver(metadata)

    # Resolve magic variables in invoice
    # Process all string values recursively
    def _resolve_recursive(obj: Any) -> Any:
        if isinstance(obj, str):
            return resolver.resolve(obj)
        elif isinstance(obj, dict):
            return {k: _resolve_recursive(v) for k, v in obj.items()}
        elif isinstance(obj, list):
            return [_resolve_recursive(item) for item in obj]
        return obj

    invoice_file.invoice_obj = _resolve_recursive(invoice_file.invoice_obj)

    # Save
    invoice_file.overwrite()
    logger.info(f"Magic variables resolved in {invoice_file_path}")
```

```python
# src/rdetoolkit/invoicefile/_rule_replacer.py
"""Rule-based filename mapping and replacement.

This module provides classes and functions for applying rule-based
transformations to filenames in invoice files.
"""
from __future__ import annotations

import re
from pathlib import Path
from typing import Any

from rdetoolkit.invoicefile._helpers import logger


class RuleBasedReplacer:
    """Rule-based string replacer with pattern matching support.

    This class applies replacement rules to strings, supporting both
    literal string replacement and regular expression patterns.

    Attributes:
        replacement_rule (dict[str, Any]): Dictionary mapping source patterns to replacement values.

    Example:
        replacer = RuleBasedReplacer({
            "old_name.txt": "new_name.txt",
            r"data_(\d+)\.csv": r"dataset_\1.csv"  # regex pattern
        })
        result = replacer.apply("data_123.csv")
        # result: "dataset_123.csv"
    """

    def __init__(self, replacement_rule: dict[str, Any]):
        """Initialize the replacer with a rule dictionary.

        Args:
            replacement_rule: Dictionary mapping source patterns to replacement values.
        """
        self.replacement_rule = replacement_rule

    def apply(self, target: str) -> str:
        """Apply replacement rules to target string.

        Args:
            target: String to apply replacements to.

        Returns:
            Transformed string after applying all matching rules.
        """
        # Implementation from lines 1094-1249
        result = target

        for pattern, replacement in self.replacement_rule.items():
            # Try literal match first
            if pattern == result:
                result = str(replacement)
                break

            # Try regex match
            try:
                regex = re.compile(pattern)
                if regex.search(result):
                    result = regex.sub(str(replacement), result)
                    break
            except re.error:
                # Not a valid regex, skip
                continue

        return result

    def apply_to_dict(
        self, target_dict: dict[str, Any], key: str = "filepath"
    ) -> dict[str, Any]:
        """Apply replacement rules to specified key in dictionary.

        Args:
            target_dict: Dictionary containing the target key.
            key: Key in dictionary to apply replacements to (default: "filepath").

        Returns:
            Modified dictionary with replacements applied.
        """
        # Implementation details
        if key in target_dict:
            target_dict[key] = self.apply(target_dict[key])
        return target_dict


def apply_default_filename_mapping_rule(
    replacement_rule: dict[str, Any], save_file_path: str | Path
) -> dict[str, Any]:
    """Apply default filename mapping rules to a file path.

    This function creates a RuleBasedReplacer and applies rules to generate
    a new filename mapping dictionary.

    Args:
        replacement_rule: Dictionary of replacement rules to apply.
        save_file_path: Target file path to apply rules to.

    Returns:
        Dictionary mapping original filename to new filename.

    Example:
        rules = {"old_prefix": "new_prefix"}
        result = apply_default_filename_mapping_rule(rules, "old_prefix_data.csv")
        # result: {"old_prefix_data.csv": "new_prefix_data.csv"}
    """
    # Implementation from lines 1418-1447
    replacer = RuleBasedReplacer(replacement_rule)

    original_name = Path(save_file_path).name
    transformed_name = replacer.apply(original_name)

    mapping = {original_name: transformed_name}

    logger.debug(f"Filename mapping applied: {original_name} -> {transformed_name}")

    return mapping
```

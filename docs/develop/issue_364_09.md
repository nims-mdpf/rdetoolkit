```python
# src/rdetoolkit/invoicefile/_smarttable.py
"""SmartTable file handling.

This module provides the SmartTableFile class for processing SmartTable
files, which are used in extended_mode for intelligent tabular data handling.
"""
from __future__ import annotations

import os
from pathlib import Path
from typing import TYPE_CHECKING, Any

from rdetoolkit.exceptions import StructuredError
from rdetoolkit.invoicefile._helpers import _ensure_pandas, logger

if TYPE_CHECKING:
    import pandas as pd


class SmartTableFile:
    """Handler for SmartTable files in extended mode.

    SmartTable files provide enhanced tabular data processing with features
    like early exit detection, dynamic column handling, and intelligent
    data type inference.

    Attributes:
        smarttable_path (Path): Path to the SmartTable file.
        df_smarttable (pd.DataFrame): DataFrame containing SmartTable data.

    Args:
        smarttable_path (Path): Path to SmartTable file (CSV or Excel).

    Example:
        smarttable = SmartTableFile(Path("smarttable.csv"))
        print(smarttable.df_smarttable.head())

        # Process with early exit detection
        results = smarttable.process_with_early_exit(exit_marker="END")
    """

    def __init__(self, smarttable_path: Path):
        """Initialize SmartTableFile.

        Args:
            smarttable_path: Path to SmartTable file.

        Raises:
            StructuredError: If file doesn't exist or cannot be read.
        """
        self.smarttable_path = Path(smarttable_path)
        self.df_smarttable = self.read()

    def read(self, *, target_path: Path | None = None) -> pd.DataFrame:
        """Read SmartTable file into DataFrame.

        Args:
            target_path: Optional path to read from (defaults to self.smarttable_path).

        Returns:
            DataFrame containing SmartTable data.

        Raises:
            StructuredError: If file doesn't exist or read fails.
        """
        if target_path is None:
            target_path = self.smarttable_path

        if not os.path.exists(target_path):
            emsg = f"ERROR: SmartTable file not found {target_path}"
            raise StructuredError(emsg)

        pd = _ensure_pandas()

        # Detect file format and read accordingly
        if target_path.suffix.lower() in [".xlsx", ".xls"]:
            df = pd.read_excel(target_path, dtype=str)
        elif target_path.suffix.lower() == ".csv":
            df = pd.read_csv(target_path, dtype=str)
        else:
            emsg = f"ERROR: Unsupported file format for SmartTable: {target_path.suffix}"
            raise StructuredError(emsg)

        logger.info(f"SmartTable loaded: {target_path} ({len(df)} rows)")
        return df

    def process_with_early_exit(
        self, exit_marker: str = "END", marker_column: int = 0
    ) -> pd.DataFrame:
        """Process SmartTable with early exit detection.

        Stops processing when exit_marker is found in the specified column.

        Args:
            exit_marker: String marker indicating end of data (default: "END").
            marker_column: Column index to check for exit marker (default: 0).

        Returns:
            DataFrame containing only rows before exit marker.

        Example:
            # SmartTable with "END" marker
            df = smarttable.process_with_early_exit(exit_marker="END")
        """
        pd = _ensure_pandas()
        df = self.df_smarttable.copy()

        # Find exit marker
        for idx, row in df.iterrows():
            if marker_column < len(row):
                cell_value = str(row.iloc[marker_column]).strip().upper()
                if cell_value == exit_marker.upper():
                    logger.info(f"Early exit marker found at row {idx}")
                    return df.iloc[:idx].copy()

        # No exit marker found, return full DataFrame
        return df

    def get_column_types(self) -> dict[str, str]:
        """Infer column types from SmartTable data.

        Returns:
            Dictionary mapping column names to inferred types.

        Example:
            types = smarttable.get_column_types()
            # {'column1': 'numeric', 'column2': 'string', ...}
        """
        pd = _ensure_pandas()
        column_types = {}

        for col in self.df_smarttable.columns:
            # Try to infer type
            try:
                pd.to_numeric(self.df_smarttable[col], errors="raise")
                column_types[col] = "numeric"
            except (ValueError, TypeError):
                column_types[col] = "string"

        return column_types

    def validate_structure(
        self, required_columns: list[str] | None = None
    ) -> tuple[bool, list[str]]:
        """Validate SmartTable structure.

        Args:
            required_columns: List of required column names (optional).

        Returns:
            Tuple of (is_valid, missing_columns).

        Example:
            valid, missing = smarttable.validate_structure(["col1", "col2"])
            if not valid:
                print(f"Missing columns: {missing}")
        """
        if required_columns is None:
            return True, []

        existing_columns = set(self.df_smarttable.columns)
        missing = [col for col in required_columns if col not in existing_columns]

        is_valid = len(missing) == 0
        return is_valid, missing

    def to_dict(self, orient: str = "records") -> list[dict[str, Any]] | dict[str, Any]:
        """Convert SmartTable to dictionary representation.

        Args:
            orient: Orientation for dictionary conversion (default: "records").
                See pandas.DataFrame.to_dict for options.

        Returns:
            Dictionary representation of SmartTable.
        """
        return self.df_smarttable.to_dict(orient=orient)

    def save(
        self, save_path: str | Path, *, format: str = "csv", index: bool = False
    ) -> None:
        """Save SmartTable to file.

        Args:
            save_path: Path where file will be saved.
            format: Output format ("csv" or "excel").
            index: Whether to write row index (default: False).

        Raises:
            StructuredError: If save fails.
        """
        pd = _ensure_pandas()
        save_path = Path(save_path)

        try:
            if format.lower() == "csv":
                self.df_smarttable.to_csv(save_path, index=index)
            elif format.lower() in ["excel", "xlsx"]:
                self.df_smarttable.to_excel(save_path, index=index)
            else:
                emsg = f"Unsupported format: {format}"
                raise StructuredError(emsg)

            logger.info(f"SmartTable saved to {save_path}")
        except Exception as e:
            emsg = f"Failed to save SmartTable: {e}"
            raise StructuredError(emsg) from e
```

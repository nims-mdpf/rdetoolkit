# src/rdetoolkit/invoicefile/_sheet_processing.py
"""Excel sheet processing utilities for invoice files.

This module provides functions for identifying and processing different
types of sheets in Excel invoice files (invoice, general term, specific term).
"""
from __future__ import annotations

import warnings
from collections.abc import Callable
from pathlib import Path
from typing import TYPE_CHECKING, Literal

from rdetoolkit.invoicefile._helpers import _ensure_pandas, logger

if TYPE_CHECKING:
    import pandas as pd

# Type alias for sheet types
SheetType = Literal["invoice", "general_term", "specific_term", "unknown"]


def read_excelinvoice(
    excelinvoice_filepath: Path | str,
) -> tuple[pd.DataFrame, pd.DataFrame | None, pd.DataFrame | None]:
    """Deprecated wrapper around :class:`ExcelInvoiceFile`.

    This helper will be removed in version 1.5.0. Please instantiate ``ExcelInvoiceFile``
    directly and use the ``dfexcelinvoice``, ``df_general``, and ``df_specific`` attributes.
    """
    from rdetoolkit.invoicefile._excelinvoice import ExcelInvoiceFile

    warnings.warn(
        "read_excelinvoice() is deprecated and will be removed in version 1.5.0. "
        "Instantiate ExcelInvoiceFile(invoice_path) and use the .dfexcelinvoice, "
        ".df_general, and .df_specific attributes instead.",
        DeprecationWarning,
        stacklevel=2,
    )
    excel_invoice = ExcelInvoiceFile(Path(excelinvoice_filepath))
    return excel_invoice.dfexcelinvoice, excel_invoice.df_general, excel_invoice.df_specific


def _process_invoice_sheet(df: pd.DataFrame) -> pd.DataFrame:
    """Process the main invoice sheet by extracting headers and data rows.

    Args:
        df: Raw DataFrame from Excel invoice sheet.

    Returns:
        Processed DataFrame with combined headers and data rows.
    """
    df = df.dropna(axis=0, how="all").dropna(axis=1, how="all")
    hd1 = list(df.iloc[1, :].fillna(""))
    hd2 = list(df.iloc[2, :].fillna(""))
    df.columns = [f"{s1}/{s2}" if s1 else s2 for s1, s2 in zip(hd1, hd2)]
    return df.iloc[4:, :].reset_index(drop=True).copy()


def _process_general_term_sheet(df: pd.DataFrame) -> pd.DataFrame:
    """Process the general term sheet by cleaning empty rows/columns.

    Args:
        df: Raw DataFrame from general term sheet.

    Returns:
        Cleaned DataFrame.
    """
    return df.dropna(axis=0, how="all").dropna(axis=1, how="all")


def _process_specific_term_sheet(df: pd.DataFrame) -> pd.DataFrame:
    """Process the specific term sheet by cleaning empty rows/columns.

    Args:
        df: Raw DataFrame from specific term sheet.

    Returns:
        Cleaned DataFrame.
    """
    return df.dropna(axis=0, how="all").dropna(axis=1, how="all")


def _identify_sheet_type(sh_name: str, df: pd.DataFrame) -> SheetType:
    """Identify the type of Excel sheet based on its name and content.

    Args:
        sh_name: Name of the Excel sheet.
        df: DataFrame containing sheet data.

    Returns:
        One of: "invoice", "general_term", "specific_term", or "unknown".
    """
    # Priority 1: Check if sheet name matches known patterns
    sh_name_lower = sh_name.lower().replace(" ", "").replace("_", "")

    if "generalterm" in sh_name_lower:
        return "general_term"
    if "specificterm" in sh_name_lower:
        return "specific_term"

    # Priority 2: Check first cell content for invoice sheet
    if not df.empty:
        first_cell = str(df.iloc[0, 0]).strip().lower()
        if first_cell in ("invoiceform", "invoice_form"):
            return "invoice"

    # Priority 3: Default to unknown
    return "unknown"


# Sheet processor registry
_SHEET_PROCESSORS: dict[SheetType, Callable[[pd.DataFrame], pd.DataFrame]] = {
    "invoice": _process_invoice_sheet,
    "general_term": _process_general_term_sheet,
    "specific_term": _process_specific_term_sheet,
}


def check_exist_rawfiles(
    dfexcelinvoice: pd.DataFrame,
    excel_rawfiles: list[Path],
) -> list[Path]:
    """Check which raw files listed in the invoice exist on disk.

    Args:
        dfexcelinvoice: Invoice DataFrame containing file listings.
        excel_rawfiles: List of raw file paths to check.

    Returns:
        List of existing raw file paths.
    """
    exist_rawfiles: list[Path] = []

    for _, row in dfexcelinvoice.iterrows():
        filepath = row.get("filepath")
        if not filepath:
            continue

        for rawfile in excel_rawfiles:
            if rawfile.name == filepath or str(rawfile).endswith(str(filepath)):
                exist_rawfiles.append(rawfile)
                break

    return exist_rawfiles


def check_exist_rawfiles_for_folder(
    dfexcelinvoice: pd.DataFrame,
    rawfiles_tpl: tuple,
) -> list:
    """Check which raw files/folders exist for folder-based invoice mode.

    Args:
        dfexcelinvoice: Invoice DataFrame.
        rawfiles_tpl: Tuple of raw file/folder paths.

    Returns:
        List of existing file/folder paths.
    """
    exist_rawfiles = []

    for _, row in dfexcelinvoice.iterrows():
        filepath = row.get("filepath")
        if not filepath:
            continue

        for rawfile in rawfiles_tpl:
            rawfile_path = Path(rawfile)
            if rawfile_path.name == filepath or str(rawfile_path).endswith(str(filepath)):
                exist_rawfiles.append(rawfile)
                break

    return exist_rawfiles
